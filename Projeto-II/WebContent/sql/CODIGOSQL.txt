CREATE DATABASE SGE;


CREATE TABLE ADMINISTRADOR(

	ID INT AUTO_INCREMENT PRIMARY KEY,
	NOME VARCHAR(50),
	SIAPE VARCHAR(10),
	SENHA VARCHAR(8)
);

CREATE TABLE PROFESSOR(

	ID INT AUTO_INCREMENT PRIMARY KEY,
	NOME VARCHAR(50) NOT NULL,
	SIAPE VARCHAR(10) NOT NULL,
	EMAIL VARCHAR(50) NOT NULL,
	SENHA VARCHAR(20),
	TELEFONE VARCHAR(14),
	CAMPUS VARCHAR(20),
	REG_TRABALHO VARCHAR(4),
	AREA VARCHAR(20),
	CLASSE VARCHAR(10),
	DIRETORIA VARCHAR(20)

);

CREATE TABLE PLANO_TRABALHO(

	ID INT AUTO_INCREMENT PRIMARY KEY,
	ATIVIDADE_APOIO VARCHAR(200),
	ATIVIDADE_PESQUISA VARCHAR(200),
	ATIVIDADE_EXTENSAO VARCHAR(200),
	ID_PROFESSOR INT,
	FOREIGN KEY(ID_PROFESSOR) REFERENCES PROFESSOR(ID)
);

CREATE TABLE ATIVIDADE_PEDAGOGICA(

	ID INT AUTO_INCREMENT PRIMARY KEY,
	ATIVIDADE VARCHAR(100),
	PORTARIA VARCHAR(10),
	ID_PLANO_TRABALHO INT,
	FOREIGN KEY(ID_PLANO_TRABALHO) REFERENCES PLANO_TRABALHO(ID)
	
);


CREATE TABLE ATIVIDADE_ENSINO(

	ID INT AUTO_INCREMENT PRIMARY KEY,
	DISCIPLINA VARCHAR(50),
	CURSO VARCHAR(40),
	CARGA_HORARIA INT,
	ID_PLANO_TRABALHO INT,
	FOREIGN KEY(ID_PLANO_TRABALHO) REFERENCES PLANO_TRABALHO(ID)

);

CREATE TABLE REGISTROENTREGA(

	ID INT AUTO_INCREMENT PRIMARY KEY,
	DATA_ENTREGA DATE,
	ID_ADMIN INT,
	ID_PROFESSOR INT,
	FOREIGN KEY(ID_ADMIN) REFERENCES ADMINISTRADOR(ID),
	FOREIGN KEY(ID_PROFESSOR) REFERENCES PROFESSOR(ID)

);

CREATE TABLE ENTREGA (
    ID INT AUTO_INCREMENT PRIMARY KEY,
	SIAPE VARCHAR(10),
	NOME VARCHAR(50),
	SEMESTRE VARCHAR(10),
	DATAE DATE
);
